import { GitHub } from "@objectiveai/function-agent";

const issueNumber = parseInt(process.argv[2], 10);
const comment = process.argv[3];

if (isNaN(issueNumber) || !comment) {
  console.error("Usage: ts-node commentOnIssue.ts <issue_number> <comment>");
  process.exit(1);
}

GitHub.commentOnIssue(issueNumber, comment);
console.log(`Commented on issue #${issueNumber}`);
